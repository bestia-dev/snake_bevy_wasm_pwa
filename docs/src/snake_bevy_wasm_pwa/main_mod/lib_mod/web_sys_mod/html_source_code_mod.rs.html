3/// HtmlSourceCode - type to manipulate HTML source code safer than with string functions only  
4/// WARNING for HTML INJECTION!   
5/// HTML is the standard markup language for Web pages. HTML source code is just a text.  
6/// It is easy to read, write, understand and parse.  
7/// The syntax of HTML source code is similar to XML structured with elements, tags, nodes, texts, comments and attributes.  
8/// The browser then transforms this HTML source code into the DOM tree and then renders that.  
9/// It is very tempting to modify this source code in our application with string manipulation and then pass it to the browser.  
10/// The html source code (it is just a string) that is provided by the programmer is always ok, he wants it to work properly.  
11/// The BIG problem arises when we need to inject some user provided data into the HTML source code.  
12/// The HTML syntax mixes instructions and data together and this creates a BIG problem.  
13/// Never put user provided strings in a html source code directly, because it can contain an HTML injection attack.  
14/// We need to encode all user data before putting it into the HTML source code.  
15/// There are 2 types of encodings: one for attributes values and another for text nodes.  
16/// We will create a new type that makes it safer and easier for the programmer to replace data in the HTML source code.  
17pub struct HtmlSourceCode {
18    html: String,
19}
20
21impl HtmlSourceCode {
22    /// The programmer provides a &'static str to initiate HtmlSourceCode.  
23    /// The html source code coming from the programmer is always ok, he wants it to work properly.  
24    /// The data that will be replaced, have a recognizable, unique and delimited value.  
25    pub fn new(html_code: &'static str) -> Self {
26        HtmlSourceCode {
27            html: html_code.to_string(),
28        }
29    }
30
31    /// get the well formed html  
32    /// We trust the programmer to carefully work with HtmlSourceCode to be always well formed and without HTML injection.  
33    pub fn get_html(&self) -> String {
34        self.html.clone()
35    }
36
37    /// This must be pure text, no html element are allowed for bold or italic...  
38    /// We trust the programmer that it will replace only the anticipated placeholders.  
39    pub fn replace_text_node(&mut self, placeholder: &'static str, text: &str) {
40        self.html = self.html.replace(placeholder, &html_escape::encode_text(text));
41    }
42
43    /// The attribute value must be double_quoted.  
44    /// We trust the programmer that it will replace only the anticipated placeholders.  
45    pub fn replace_attribute_value(&mut self, placeholder: &'static str, value: &str) {
46        self.html = self.html.replace(placeholder, &html_escape::encode_double_quoted_attribute(value));
47    }
48
49    /// We expect the HtmlSourceCode to be well formed. For that we trust the programmer.  
50    /// We trust the programmer that it will replace only the anticipated placeholders.  
51    pub fn replace_html_source_code(&mut self, placeholder: &'static str, html_source_code: &HtmlSourceCode) {
52        self.html = self.html.replace(placeholder, &html_source_code.get_html());
53    }
54
55    /// Injects the HTMLSourceCode into a DOM element.  
56    /// We trust the programmer to carefully work with HtmlSourceCode to be always well formed and without HTML injection.  
57    pub fn inject_into_dom_element(self, element_id: &str) {
58        let html_element = super::get_element_by_id(element_id);
59        html_element.set_inner_html(&self.html);
60    }
61}
