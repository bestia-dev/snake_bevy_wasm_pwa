21/// return the global window object  
22pub fn window() -> web_sys::Window {
23    unwrap!(web_sys::window())
24}
25
26/// get element by id  
27pub fn get_element_by_id(element_id: &str) -> web_sys::Element {
28    let document = unwrap!(window().document());
29    unwrap!(document.get_element_by_id(element_id))
30}
31
32/// get html element by id  
33pub fn get_html_element_by_id(element_id: &str) -> web_sys::HtmlElement {
34    let element = get_element_by_id(element_id);
35    let html_element: web_sys::HtmlElement = unwrap!(element.dyn_into::<web_sys::HtmlElement>());
36    //return
37    html_element
38}

40/// get input element value string by id  
41pub fn get_input_element_value_string_by_id(element_id: &str) -> String {
42    // debug_write("before get_element_by_id");
43    let input_element = get_element_by_id(element_id);
44    // debug_write("before dyn_into");
45    let input_html_element = unwrap!(input_element.dyn_into::<web_sys::HtmlInputElement>());
46    // debug_write("before value()");
47    input_html_element.value()
48}

50/// add event listener for button  
51pub fn add_listener_to_button(element_id: &str, fn_on_click_button: &'static (dyn Fn() + 'static)) {
52    let handler_1 = Box::new(move || {
53        fn_on_click_button();
54    }) as Box<dyn FnMut()>;
55    let closure = Closure::wrap(handler_1);
56
57    let html_element = get_html_element_by_id(element_id);
58    html_element.set_onclick(Some(closure.as_ref().unchecked_ref()));
59    closure.forget();
60}

62/// add event listener for onhashchange  
63pub fn add_listener_for_onhashchange(fn_on_hash_change: &'static (dyn Fn() + 'static)) {
64    let handler_1 = Box::new(move || {
65        fn_on_hash_change();
66    }) as Box<dyn FnMut()>;
67    let closure = Closure::wrap(handler_1);
68
69    window().set_onhashchange(Some(closure.as_ref().unchecked_ref()));
70    closure.forget();
71}

73/// set inner text

75    let html_element = get_html_element_by_id(element_id);

79/// open URL in same tab (PWA don't have tabs, only one windows)  
80pub fn open_url(url: &str) {
81    dbg!(url);
82    window().location().assign(url).unwrap();
83    // Strange behavior: if url has hash, then it does not load ?!?
84    match window().location().hash() {
85        Ok(hash) => {
86            dbg!(&hash);
87            window().location().set_hash(&hash).unwrap();
88        }
89        Err(_err) => {}
90    }
91}
