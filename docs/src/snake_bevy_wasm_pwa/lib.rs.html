74use wasm_bindgen::prelude::*;
75
76mod main_mod;
77/// LibraryError must be accessible in every module.
78pub use main_mod::LibraryError;

80#[wasm_bindgen(start)]
81/// To start the Wasm application, wasm_bindgen runs this functions
82pub fn wasm_bindgen_start() -> Result<(), JsValue> {
83    // Initialize debugging for when/if something goes wrong.
84    console_error_panic_hook::set_once();
85    // write the app version just for debug purposes
86    dbg!("snake_bevy_wasm_pwa v{}", env!("CARGO_PKG_VERSION"));
87
88    main_mod::main();
89    // return
90    Ok(())
91}
